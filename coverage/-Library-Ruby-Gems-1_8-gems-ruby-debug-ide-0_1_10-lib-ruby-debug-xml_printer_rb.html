 95     def print_variable(name, value, kind)
 96       unless value
 97         print("<variable name=\"%s\" kind=\"%s\"/>", CGI.escapeHTML(name), kind)
 98         return
 99       end
100       if value.is_a?(Array) || value.is_a?(Hash)
101         has_children = !value.empty?
102         unless has_children
103           value_str = "Empty #{value.class}"
104         else
105           value_str = "#{value.class} (#{value.size} element(s))"
106         end
107       else
108         has_children = !value.instance_variables.empty? || !value.class.class_variables.empty?
109         value_str = value.to_s || 'nil'
110         unless value_str.is_a?(String)
111           value_str = "ERROR: #{value.class}.to_s method returns #{value_str.class}. Should return String." 
112         end
113         if value_str =~ /^\"(.*)"$/
114           value_str = $1
115         end
116       end
117       value_str = "[Binary Data]" if value_str.is_binary_data?
118       print("<variable name=\"%s\" kind=\"%s\" value=\"%s\" type=\"%s\" hasChildren=\"%s\" objectId=\"%#+x\"/>",
119           CGI.escapeHTML(name), kind, CGI.escapeHTML(value_str), value.class,
120           has_children, value.respond_to?(:object_id) ? value.object_id : value.id)
121     end
