867 # Needs to be duplicated from Active Support since its needed before Active
868 # Support is available. Here both Options and Hash are namespaced to prevent
869 # conflicts with other implementations AND with the classes residing in Active Support.
870 class Rails::OrderedOptions < Array #:nodoc:
871   def []=(key, value)
872     key = key.to_sym
873 
874     if pair = find_pair(key)
875       pair.pop
876       pair << value
877     else
878       self << [key, value]
879     end
880   end
881 
882   def [](key)
883     pair = find_pair(key.to_sym)
884     pair ? pair.last : nil
885   end

887   def method_missing(name, *args)
888     if name.to_s =~ /(.*)=$/
889       self[$1.to_sym] = args.first
