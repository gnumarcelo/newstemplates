 12       # Returns the URL for the set of +options+ provided. This takes the
 13       # same options as +url_for+ in Action Controller (see the
 14       # documentation for ActionController::Base#url_for). Note that by default
 15       # <tt>:only_path</tt> is <tt>true</tt> so you'll get the relative /controller/action
 16       # instead of the fully qualified URL like http://example.com/controller/action.
 17       #
 18       # When called from a view, url_for returns an HTML escaped url. If you
 19       # need an unescaped url, pass <tt>:escape => false</tt> in the +options+.
 20       #
 21       # ==== Options
 22       # * <tt>:anchor</tt> - Specifies the anchor name to be appended to the path.
 23       # * <tt>:only_path</tt> - If true, returns the relative URL (omitting the protocol, host name, and port) (<tt>true</tt> by default unless <tt>:host</tt> is specified).
 24       # * <tt>:trailing_slash</tt> - If true, adds a trailing slash, as in "/archive/2005/". Note that this
 25       #   is currently not recommended since it breaks caching.
 26       # * <tt>:host</tt> - Overrides the default (current) host if provided.
 27       # * <tt>:protocol</tt> - Overrides the default (current) protocol if provided.
 28       # * <tt>:user</tt> - Inline HTTP authentication (only plucked out if <tt>:password</tt> is also present).
 29       # * <tt>:password</tt> - Inline HTTP authentication (only plucked out if <tt>:user</tt> is also present).
 30       # * <tt>:escape</tt> - Determines whether the returned URL will be HTML escaped or not (<tt>true</tt> by default).
 31       #
 32       # ==== Relying on named routes
 33       #
 34       # If you instead of a hash pass a record (like an Active Record or Active Resource) as the options parameter,
 35       # you'll trigger the named route for that record. The lookup will happen on the name of the class. So passing
 36       # a Workshop object will attempt to use the workshop_path route. If you have a nested route, such as 
 37       # admin_workshop_path you'll have to call that explicitly (it's impossible for url_for to guess that route). 
 38       #
 39       # ==== Examples
 40       #   <%= url_for(:action => 'index') %>
 41       #   # => /blog/
 42       #
 43       #   <%= url_for(:action => 'find', :controller => 'books') %>
 44       #   # => /books/find
 45       #
 46       #   <%= url_for(:action => 'login', :controller => 'members', :only_path => false, :protocol => 'https') %>
 47       #   # => https://www.railsapplication.com/members/login/
 48       #
 49       #   <%= url_for(:action => 'play', :anchor => 'player') %>
 50       #   # => /messages/play/#player
 51       #
 52       #   <%= url_for(:action => 'checkout', :anchor => 'tax&ship') %>
 53       #   # => /testing/jump/#tax&amp;ship
 54       #
 55       #   <%= url_for(:action => 'checkout', :anchor => 'tax&ship', :escape => false) %>
 56       #   # => /testing/jump/#tax&ship
 57       #
 58       #   <%= url_for(Workshop.new) %>
 59       #   # relies on Workshop answering a new_record? call (and in this case returning true)
 60       #   # => /workshops
 61       #
 62       #   <%= url_for(@workshop) %>
 63       #   # calls @workshop.to_s
 64       #   # => /workshops/5
 65       def url_for(options = {})
 66         case options
 67         when Hash
 68           show_path =  options[:host].nil? ? true : false
 69           options = { :only_path => show_path }.update(options.symbolize_keys)
 70           escape  = options.key?(:escape) ? options.delete(:escape) : true
 71           url     = @controller.send(:url_for, options)
 72         when String
 73           escape = true
 74           url    = options
 75         when NilClass
 76           url = @controller.send(:url_for, nil)
 77         else
 78           escape = false
 79           url    = polymorphic_path(options)
 80         end
 81 
 82         escape ? escape_once(url) : url
 83       end

385       # Creates a mailto link tag to the specified +email_address+, which is
386       # also used as the name of the link unless +name+ is specified. Additional
387       # HTML attributes for the link can be passed in +html_options+.
388       #
389       # mail_to has several methods for hindering email harvesters and customizing
390       # the email itself by passing special keys to +html_options+.
391       #
392       # ==== Options
393       # * <tt>:encode</tt>  - This key will accept the strings "javascript" or "hex".
394       #   Passing "javascript" will dynamically create and encode the mailto: link then
395       #   eval it into the DOM of the page. This method will not show the link on
396       #   the page if the user has JavaScript disabled. Passing "hex" will hex
397       #   encode the +email_address+ before outputting the mailto: link.
398       # * <tt>:replace_at</tt>  - When the link +name+ isn't provided, the
399       #   +email_address+ is used for the link label. You can use this option to
400       #   obfuscate the +email_address+ by substituting the @ sign with the string
401       #   given as the value.
402       # * <tt>:replace_dot</tt>  - When the link +name+ isn't provided, the
403       #   +email_address+ is used for the link label. You can use this option to
404       #   obfuscate the +email_address+ by substituting the . in the email with the
405       #   string given as the value.
406       # * <tt>:subject</tt>  - Preset the subject line of the email.
407       # * <tt>:body</tt> - Preset the body of the email.
408       # * <tt>:cc</tt>  - Carbon Copy addition recipients on the email.
409       # * <tt>:bcc</tt>  - Blind Carbon Copy additional recipients on the email.
410       #
411       # ==== Examples
412       #   mail_to "me@domain.com" 
413       #   # => <a href="mailto:me@domain.com">me@domain.com</a>
414       #
415       #   mail_to "me@domain.com", "My email", :encode => "javascript"  
416       #   # => <script type="text/javascript">eval(unescape('%64%6f%63...%6d%65%6e'))</script>
417       #
418       #   mail_to "me@domain.com", "My email", :encode => "hex"  
419       #   # => <a href="mailto:%6d%65@%64%6f%6d%61%69%6e.%63%6f%6d">My email</a>
420       #
421       #   mail_to "me@domain.com", nil, :replace_at => "_at_", :replace_dot => "_dot_", :class => "email"  
422       #   # => <a href="mailto:me@domain.com" class="email">me_at_domain_dot_com</a>
423       #
424       #   mail_to "me@domain.com", "My email", :cc => "ccaddress@domain.com",
425       #            :subject => "This is an example email"  
426       #   # => <a href="mailto:me@domain.com?cc=ccaddress@domain.com&subject=This%20is%20an%20example%20email">My email</a>
427       def mail_to(email_address, name = nil, html_options = {})
428         html_options = html_options.stringify_keys
429         encode = html_options.delete("encode").to_s
430         cc, bcc, subject, body = html_options.delete("cc"), html_options.delete("bcc"), html_options.delete("subject"), html_options.delete("body")
431 
432         string = ''
433         extras = ''
434         extras << "cc=#{CGI.escape(cc).gsub("+", "%20")}&" unless cc.nil?
435         extras << "bcc=#{CGI.escape(bcc).gsub("+", "%20")}&" unless bcc.nil?
436         extras << "body=#{CGI.escape(body).gsub("+", "%20")}&" unless body.nil?
437         extras << "subject=#{CGI.escape(subject).gsub("+", "%20")}&" unless subject.nil?
438         extras = "?" << extras.gsub!(/&?$/,"") unless extras.empty?
439 
440         email_address = email_address.to_s
441 
442         email_address_obfuscated = email_address.dup
443         email_address_obfuscated.gsub!(/@/, html_options.delete("replace_at")) if html_options.has_key?("replace_at")
444         email_address_obfuscated.gsub!(/\./, html_options.delete("replace_dot")) if html_options.has_key?("replace_dot")
445 
446         if encode == "javascript"
447           "document.write('#{content_tag("a", name || email_address, html_options.merge({ "href" => "mailto:"+email_address+extras }))}');".each_byte do |c|
448             string << sprintf("%%%x", c)
449           end
450           "<script type=\"#{Mime::JS}\">eval(unescape('#{string}'))</script>"
451         elsif encode == "hex"
452           email_address_encoded = ''
453           email_address_obfuscated.each_byte do |c|
454             email_address_encoded << sprintf("&#%d;", c)
455           end
456 
457           protocol = 'mailto:'
458           protocol.each_byte { |c| string << sprintf("&#%d;", c) }
459 
460           email_address.each_byte do |c|
461             char = c.chr
462             string << (char =~ /\w/ ? sprintf("%%%x", c) : char)
463           end
464           content_tag "a", name || email_address_encoded, html_options.merge({ "href" => "#{string}#{extras}" })
465         else
466           content_tag "a", name || email_address_obfuscated, html_options.merge({ "href" => "mailto:#{email_address}#{extras}" })

470       # True if the current request URI was generated by the given +options+.
471       #
472       # ==== Examples
473       # Let's say we're in the <tt>/shop/checkout</tt> action.
474       #
475       #   current_page?(:action => 'process')
476       #   # => false
477       #
478       #   current_page?(:controller => 'shop', :action => 'checkout')
479       #   # => true

481       #   current_page?(:action => 'checkout')
482       #   # => true

484       #   current_page?(:controller => 'library', :action => 'checkout')
485       #   # => false
486       def current_page?(options)
487         url_string = CGI.escapeHTML(url_for(options))
488         request = @controller.request
489         if url_string =~ /^\w+:\/\//
490           url_string == "#{request.protocol}#{request.host_with_port}#{request.request_uri}"
491         else
492           url_string == request.request_uri
493         end
494       end

497         def convert_options_to_javascript!(html_options, url = '')
498           confirm, popup = html_options.delete("confirm"), html_options.delete("popup")
499 
500           method, href = html_options.delete("method"), html_options['href']
501 
502           html_options["onclick"] = case
503             when popup && method
504               raise ActionView::ActionViewError, "You can't use :popup and :method in the same link"
505             when confirm && popup
506               "if (#{confirm_javascript_function(confirm)}) { #{popup_javascript_function(popup)} };return false;"
507             when confirm && method
508               "if (#{confirm_javascript_function(confirm)}) { #{method_javascript_function(method)} };return false;"
509             when confirm

511             when method
512               "#{method_javascript_function(method, url, href)}return false;"
513             when popup

517           end

528         def method_javascript_function(method, url = '', href = nil)
529           action = (href && url.size > 0) ? "'#{url}'" : 'this.href'
530           submit_function =
531             "var f = document.createElement('form'); f.style.display = 'none'; " +
532             "this.parentNode.appendChild(f); f.method = 'POST'; f.action = #{action};"
533 
534           unless method == :post
535             submit_function << "var m = document.createElement('input'); m.setAttribute('type', 'hidden'); "
536             submit_function << "m.setAttribute('name', '_method'); m.setAttribute('value', '#{method}'); f.appendChild(m);"
537           end
538 
539           if protect_against_forgery?
540             submit_function << "var s = document.createElement('input'); s.setAttribute('type', 'hidden'); "
541             submit_function << "s.setAttribute('name', '#{request_forgery_protection_token}'); s.setAttribute('value', '#{escape_javascript form_authenticity_token}'); f.appendChild(s);"
542           end
543           submit_function << "f.submit();"

546         # Processes the _html_options_ hash, converting the boolean
547         # attributes from true/false form into the form required by
548         # HTML/XHTML.  (An attribute is considered to be boolean if
549         # its name is listed in the given _bool_attrs_ array.)
550         #
551         # More specifically, for each boolean attribute in _html_options_
552         # given as:
553         #
554         #     "attr" => bool_value
555         #
556         # if the associated _bool_value_ evaluates to true, it is
557         # replaced with the attribute's name; otherwise the attribute is
558         # removed from the _html_options_ hash.  (See the XHTML 1.0 spec,
559         # section 4.5 "Attribute Minimization" for more:
560         # http://www.w3.org/TR/xhtml1/#h-4.5)
561         #
562         # Returns the updated _html_options_ hash, which is also modified
563         # in place.
564         #
565         # Example:
566         #
567         #   convert_boolean_attributes!( html_options,
568         #                                %w( checked disabled readonly ) )
569         def convert_boolean_attributes!(html_options, bool_attrs)
570           bool_attrs.each { |x| html_options[x] = x if html_options.delete(x) }
571           html_options
572         end
